CREATE TABLE `productos` (
	`id` INT(11) NOT NULL AUTO_INCREMENT,
	`nombre` VARCHAR(255) NOT NULL COLLATE 'utf8_general_ci',
	`descripcion` VARCHAR(255) NULL DEFAULT NULL COLLATE 'utf8_general_ci',
	`precio_venta` INT(11) NULL DEFAULT NULL,
	`precio_compra` INT(11) NULL DEFAULT NULL,
	`estado` ENUM('activo','inactivo') NULL DEFAULT NULL COLLATE 'utf8_general_ci',
	PRIMARY KEY (`id`) USING BTREE
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=18
;
